# danboard-cli

#### 目录介绍

    ├── bin                         执行文件目录
    │   └── www                     执行文件
    ├── src                         源码
    │   ├── assets                  静态资源
    │   │   ├── geojson             地图资源
    │   │   │   └── provinces       中国省的地图资源
    │   │   └── icons               图标资源
    │   ├── components              组件目录
    │   │   ├── charts              图表组件
    │   │   └── commons             通用业务组件
    │   ├── directive               vue 指令
    │   ├── filters                 vue 过滤器
    │   ├── mixins                  vue mixins
    │   ├── plugin                  vue 插件
    │   ├── router                  vue 路由
    │   ├── socket.io               socket 插件
    │   ├── store                   vue vuex(状态管理)
    │   │   ├── modules             vuex模块
    │   │   └── plugins             vuex持久化
    │   ├── styles                  全局样式
    │   ├── util                    工具方法
    │   └── views                   页面
    │        ├── layout             布局
    │        └── demo               demo示例(图表、组件、表格、表单)
    └── static                      (预留)
